N = int(input())

    child, parent = map(int, input().split())

    arr[child].append(parent)
    
visited = [False] * (N+1)

q = deque([[start, 0]])
visited[start] = True

while q:
    now, count = q.popleft()
    if now == end:
        break
    
    for next in arr[now]:
        if not visited[next]:
            q.append([next, count+1])
            visited[next] = True

if not visited[end]:
    print(-1)
else:
    print(count)

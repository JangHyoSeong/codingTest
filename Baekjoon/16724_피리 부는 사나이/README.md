def find(parent, x):
    if parent[x] != x:
        parent[x] = find(parent, parent[x])
    return parent[x]

def union(parent, x, y):
    root_x = find(parent, x)
    root_y = find(parent, y)
    
    if root_x != root_y:
        parent[root_y] = root_x
    


N, M = map(int, input().split())
table = [list(input()) for _ in range(N)]

parent = [i for i in range(N * M)]

move = {
    "D" : (1, 0),
    "U" : (-1, 0),
    "L" : (0, -1),
    "R" : (0, 1)
}

for i in range(N):
    for j in range(M):
        ni, nj = i + move[table[i][j]][0], j + move[table[i][j]][1]
        union(parent, i * M + j, ni * M + nj)

roots = set(find(parent, i) for i in range(N * M))
print(len(roots))

import sys

N, M = map(int, input().split())
mList = []
for i in range(N):
    mList.append(sys.stdin.readline().rstrip())

visited = [ [0] * M for i in range(N) ]
cnt = 0
ans = 0
for i in range(N):
    for j in range(M):
        if visited[i][j] == 0:
            cnt += 1
            x, y = j, i
            while True:
                if visited[y][x] != 0:
                    if visited[y][x] == cnt:
                        ans += 1
                    break

                visited[y][x] = cnt

                if mList[y][x] == "D":
                    y += 1

                elif mList[y][x] == "U":
                    y -= 1

                elif mList[y][x] == "L":
                    x -= 1

                elif mList[y][x] == "R":
                    x += 1

print(ans)

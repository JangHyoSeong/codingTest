def solution():
    import sys
    N = int(sys.stdin.readline())
    cards = [list(map(int, sys.stdin.readline().split())) for _ in range(N)]
    p, result = 0, 0
    for idx, card in enumerate(cards):
        for i in range(len(card)):
            for j in range(i+1, len(card)):
                calc = (sum(card) - card[i] - card[j]) % 10
                if calc >= result:
                    result = calc
                    p = idx
    print(p+1)
    
solution()

def get_lines_cnt(arr, N):
  memo = [1 for _ in range(N)] 

  for i in range(N):
    for j in range(i):
      if arr[i][1] > arr[j][1]:
        memo[i] = max(memo[i], memo[j] + 1)

  return max(memo)

if __name__ == '__main__':
  N = int(input())
  lines = [list(map(int, input().split())) for _ in range(N)]
  lines.sort()
  cnt = get_lines_cnt(lines, N)
  print(f'{N - cnt}\n')

from collections import Counter
def main():
    S = open(0).read().split()
    m = int(S[1])
    counter = Counter(word for word in S[2:] if len(word) >= m)
    words = list(counter)
    words.sort()
    words.sort(key=len, reverse=True)
    words.sort(key=counter.get, reverse=True)
    print("\n".join(words))

if __name__ == "__main__":
    main()

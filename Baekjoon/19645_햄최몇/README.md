            if dp[i][j]:
                dp[i+burger][j] = True
                dp[i][j+burger] = 1

    for j in range(sum_burgers + 1):
        if not dp[i][j]:
            continue

        k = sum_burgers - (i+j)
        min_util = min(k, i, j)
        max_util = max(max_util, min_util)

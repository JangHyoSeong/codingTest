w, h = map(int, input().split())

    dp[1][2][0][0] = 1

    dp[2][1][1][0] = 1

for y in range(1, h+1):
    for x in range(1, w+1):
        for dir in range(2):
            for turn in range(2):
                val = dp[y][x][dir][turn]

                    continue

                if dir == 0:
                    if x + 1 <= w:

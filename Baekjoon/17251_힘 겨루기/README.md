    left_max = max(left_max, arr[i])
    right_max = prefix_max[i+1]    

    if left_max > right_max:
        left_win += 1
    
    elif left_max < right_max:
        right_win += 1


if left_win > right_win:
    print("R")

elif left_win < right_win:
    print("B")

else:
